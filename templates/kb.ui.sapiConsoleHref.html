The directive creates a link to Storage Console in the current project. It uses kb.config to store the projectBaseUrl property.

View

    <kb-sapi-console-href path="/tables/in.c-main.table">
      <span class="btn btn-default btn-xs">
          <i class="fa fa-fw fa-table"></i>
      </span>
    </kb-sapi-console-href>

app.coffee

    angular.run([
        "kbc.application.config"
        "kb.config"
        (appConfig, kbConfig) ->
            kbConfig.projectBaseUrl = appConfig.projectBaseUrl
    ])
